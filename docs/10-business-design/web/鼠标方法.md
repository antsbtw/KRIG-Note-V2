async function clickCopy() {
  const webview = document.querySelector('webview');
  const coords = await webview.executeJavaScript(`
    (function() {
      const iframe = document.querySelector('iframe[src*="claudemcpcontent"], iframe[src*="claudeusercontent"]');
      let card = iframe;
      for (let i = 0; i < 3; i++) card = card.parentElement;
      const r = card.getBoundingClientRect();
      return { cx: Math.round(r.x + r.width/2), cy: Math.round(r.y + r.height/2),
               trX: Math.round(r.x + r.width - 30), trY: Math.round(r.y + 30) };
    })()
  `);

  // hover 触发菜单
  await window.viewAPI.wbSendMouse([
    { type: 'mouseMoved', x: coords.cx - 400, y: coords.cy, button: 'none', buttons: 0 },
    { type: 'mouseMoved', x: coords.cx, y: coords.cy, button: 'none', buttons: 0 },
    { type: 'mouseMoved', x: coords.trX, y: coords.trY, button: 'none', buttons: 0 },
  ]);
  // 稍等菜单渲染
  await new Promise(r => setTimeout(r, 250));

  // 估算 Copy to clipboard 位置 (菜单从 "..." 按钮下方展开，向左)
  // 从你截图看，菜单宽约 180px、高约 3 项 × 36px
  // 第一项 "Copy to clipboard" 大约在菜单顶部
  const copyX = coords.trX - 80;       // 菜单向左展开
  const copyY = coords.trY + 45;       // 菜单在 "..." 按钮下方 ~45px

  console.log('attempting click at', copyX, copyY);
  // 先移到菜单项位置 (保持菜单不关)，再点击
  await window.viewAPI.wbSendMouse([
    { type: 'mouseMoved', x: copyX, y: copyY, button: 'none', buttons: 0 },
  ]);
  await new Promise(r => setTimeout(r, 100));
  await window.viewAPI.wbSendMouse([
    { type: 'mousePressed',  x: copyX, y: copyY, button: 'left', buttons: 1, clickCount: 1 },
    { type: 'mouseReleased', x: copyX, y: copyY, button: 'left', buttons: 0, clickCount: 1 },
  ]);
  await new Promise(r => setTimeout(r, 600));

  const text = await window.viewAPI.aiReadClipboard();
  console.log('clipboard len:', text?.length);
  console.log('preview:', text?.slice(0, 400));
}

async function extractArtifactImage() {
  const webview = document.querySelector('webview');

  // 0. 清空剪贴板（排除上次残留）
  await webview.executeJavaScript(`navigator.clipboard.writeText('').catch(()=>{})`);

  // 1. 滚动 artifact 卡片进入视口，然后读 rect
  const coords = await webview.executeJavaScript(`
    (async function() {
      const iframe = document.querySelector('iframe[src*="claudemcpcontent"], iframe[src*="claudeusercontent"]');
      if (!iframe) return { error: 'no iframe' };
      let card = iframe;
      for (let i = 0; i < 3; i++) card = card.parentElement;
      card.scrollIntoView({ block: 'center', behavior: 'instant' });
      await new Promise(r => setTimeout(r, 400));
      const r = card.getBoundingClientRect();
      return {
        cx: Math.round(r.x + r.width/2),
        cy: Math.round(r.y + r.height/2),
        trX: Math.round(r.x + r.width - 30),
        trY: Math.round(r.y + 30),
        rect: { x: Math.round(r.x), y: Math.round(r.y), w: Math.round(r.width), h: Math.round(r.height) },
      };
    })()
  `);
  console.log('coords after scroll:', coords);
  if (coords.error) return;

  // 2. hover 轨迹（你验证过的）
  await window.viewAPI.wbSendMouse([
    { type: 'mouseMoved', x: coords.cx - 400, y: coords.cy, button: 'none', buttons: 0 },
    { type: 'mouseMoved', x: coords.cx, y: coords.cy, button: 'none', buttons: 0 },
    { type: 'mouseMoved', x: coords.trX, y: coords.trY, button: 'none', buttons: 0 },
  ]);
  await new Promise(r => setTimeout(r, 250));

  // 3. 点击 Copy to clipboard（估算坐标）
  const copyX = coords.trX - 80;
  const copyY = coords.trY + 45;
  await window.viewAPI.wbSendMouse([
    { type: 'mouseMoved', x: copyX, y: copyY, button: 'none', buttons: 0 },
  ]);
  await new Promise(r => setTimeout(r, 100));
  await window.viewAPI.wbSendMouse([
    { type: 'mousePressed',  x: copyX, y: copyY, button: 'left', buttons: 1, clickCount: 1 },
    { type: 'mouseReleased', x: copyX, y: copyY, button: 'left', buttons: 0, clickCount: 1 },
  ]);
  await new Promise(r => setTimeout(r, 700));

  // 4. 读图片
  const img = await window.viewAPI.wbReadClipboardImage();
  console.log('image:', img);
  if (img.success) {
    const w = window.open('');
    w.document.write('<img src="' + img.dataUrl + '" style="max-width:100%">');
  }

  // 5. 鼠标移走，避免菜单堵着
  await window.viewAPI.wbSendMouse([{ type: 'mouseMoved', x: 50, y: 50, button: 'none', buttons: 0 }]);
}
