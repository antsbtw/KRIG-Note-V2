## 3. driver 升级详细

### 3.1 plugins/build-slash-plugin.ts(Q6=A — PM Plugin 触发)

```ts
import { Plugin, PluginKey } from 'prosemirror-state';
import { slashMenuController } from '@slot/triggers/slash-menu-controller';

const slashKey = new PluginKey('slash');

interface SlashState {
  active: boolean;
  triggerPos: number;       // `/` 字符在 doc 中的位置
  query: string;
}

export function buildSlashPlugin(viewId: string): Plugin {
  return new Plugin<SlashState>({
    key: slashKey,
    state: {
      init: () => ({ active: false, triggerPos: -1, query: '' }),
      apply(tr, prev, _old, newState) {
        // 检测光标位置 + 文本判断
        const sel = newState.selection;
        if (!sel.empty) return { active: false, triggerPos: -1, query: '' };

        // 从 prev.triggerPos 到 sel.from 之间是 query
        if (prev.active) {
          // 已激活:更新 query
          const queryFrom = prev.triggerPos + 1;
          const queryTo = sel.from;
          if (queryTo < queryFrom) return { active: false, triggerPos: -1, query: '' };
          const text = newState.doc.textBetween(queryFrom, queryTo, '\n');
          if (text.includes('\n') || text.includes(' ')) {
            // query 中含空格或换行 → 关闭
            return { active: false, triggerPos: -1, query: '' };
          }
          return { ...prev, query: text };
        }

        // 未激活:检测刚输入了 `/`
        if (tr.steps.length > 0) {
          const $from = sel.$from;
          // 行首 / 段首位置才触发
          const charBefore = newState.doc.textBetween(Math.max(0, sel.from - 1), sel.from);
          if (charBefore === '/') {
            // 检查 / 前面是行首
            const beforeBefore = newState.doc.textBetween(Math.max(0, sel.from - 2), sel.from - 1);
            const isLineStart = sel.from - 1 === $from.start($from.depth) || beforeBefore === '\n' || beforeBefore === '';
            if (isLineStart) {
              return { active: true, triggerPos: sel.from - 1, query: '' };
            }
          }
        }
        return { active: false, triggerPos: -1, query: '' };
      },
    },
    view: () => ({
      update(view, _prevState) {
        const slashState = slashKey.getState(view.state);
        if (slashState?.active) {
          // 计算屏幕坐标
          const coords = view.coordsAtPos(slashState.triggerPos);
          slashMenuController.show(coords.left, coords.bottom + 4, viewId, slashState.query);
        } else {
          if (slashMenuController.getState().visible) slashMenuController.hide();
        }
      },
      destroy() {
        if (slashMenuController.getState().visible) slashMenuController.hide();
      },
    }),
    props: {
      // Esc / 上下/Enter 当 slash menu 激活时由 trigger hook 监听
    },
  });
}

/** 命令 helper:slash 选中后清除 / 跟 query,然后执行命令 */
export function clearSlashQuery(view: EditorView): void {
  const state = slashKey.getState(view.state);
  if (!state?.active) return;
  view.dispatch(view.state.tr.delete(state.triggerPos, view.state.selection.from));
}
```

driver Host mount 时把 `buildSlashPlugin(config.instanceId)` 加入 plugins 列表。

### 3.3 floating-toolbar-source.ts(driver 内监听 selection)

```ts
import { selection } from '@capabilities/selection';
import { floatingToolbarController } from '@slot/triggers/floating-toolbar-controller';

let unsubSubscription: (() => void) | null = null;

export function setupFloatingToolbarTrigger(view: EditorView, viewId: string, instanceId: string): () => void {
  // 订阅 selection capability,非空选区(text 类型)→ show,空选区 → hide
  const unsub = selection.subscribe((payload) => {
    if (payload.source !== `text-editing-driver:${instanceId}`) return;
    if (payload.kind !== 'text') return;
    if (payload.isEmpty) {
      floatingToolbarController.hide();
      return;
    }
    // 计算选区屏幕坐标(选区上方居中)
    const sel = view.state.selection;
    const fromCoords = view.coordsAtPos(sel.from);
    const toCoords = view.coordsAtPos(sel.to);
    const x = (fromCoords.left + toCoords.right) / 2;
    const y = Math.min(fromCoords.top, toCoords.top) - 8; // 选区上方 8px
    floatingToolbarController.show(x, y, viewId);
  });
  return unsub;
}
```

### 3.4 dnd 真消费 — driver 注册 dropTarget

```ts
// capability-integrations/dnd-targets.ts(L5-A 占位 → L5-B3.1 真实施)
import { dnd } from '@capabilities/drag-and-drop';

export function registerDropTargets(instanceId: string): () => void {
  const targetId = `text-editing-driver:${instanceId}:doc`;
  dnd.registerDropTarget({
    id: targetId,
    accepts: ['block'],
    computeDropPoint(coords, viewArg) {
      const view = viewArg as EditorView;
      const pos = view.posAtCoords({ left: coords.x, top: coords.y });
      if (!pos) return null;
      // 把 pos 落到最近的 block 边界
      const $pos = view.state.doc.resolve(pos.pos);
      const blockPos = $pos.before(1); // 顶层 block 起点
      return { pos: blockPos, valid: true };
    },
    onDrop({ source, target }) {
      // 移动 block:删源位 + 插目标位
      // 实施:driver api.moveBlock(instanceId, fromPos, toPos)
      const data = (source as { data: { fromPos: number; blockNode: PMNode } }).data;
      // 跳过自我拖到自我
      if (data.fromPos === target.pos) return;
      // ...
    },
  });
  return () => dnd.unregisterDropTarget(targetId);
}
```

## 4. 框架层 trigger hooks 实施

### 4.1 use-slash-trigger.ts

L5-B3.1 的实际触发由 driver PM Plugin 在 view.update 内调 `slashMenuController.show()`(§ 3.1)— **不需要框架层 hook**。

但保留 hook 接口让 view 在 NoteView 顶层 mount 时挂全局 keyboard 监听(slash menu 激活时 ↑↓ Enter Esc):

```ts
// src/slot/triggers/use-slash-trigger.ts
import { useEffect, RefObject } from 'react';
import { slashMenuController } from './slash-menu-controller';
import { commandRegistry } from '../command-registry/command-registry';
import { slashRegistry } from '../interaction-registries/slash-registry/slash-registry';

export function useSlashTrigger(elementRef: RefObject<HTMLElement | null>, viewId: string | null): void {
  useEffect(() => {
    const el = elementRef.current;
    if (!el || !viewId) return;

    let selectedIdx = 0;
    const onKeyDown = (e: KeyboardEvent) => {
      const state = slashMenuController.getState();
      if (!state.visible || state.viewId !== viewId) return;
      const items = slashRegistry.getItemsForView(viewId, state.query);
      if (e.key === 'ArrowDown') {
        e.preventDefault();
        selectedIdx = (selectedIdx + 1) % items.length;
        slashMenuController.updateQuery(state.query); // 触发重渲(L4 binding 不监听 selectedIdx — 暂略)
      } else if (e.key === 'ArrowUp') { /* ... */ }
      else if (e.key === 'Enter') {
        e.preventDefault();
        const item = items[selectedIdx];
        if (item) commandRegistry.execute(item.command);
        slashMenuController.hide();
      } else if (e.key === 'Escape') {
        slashMenuController.hide();
      }
    };
    el.addEventListener('keydown', onKeyDown);
    return () => el.removeEventListener('keydown', onKeyDown);
  }, [elementRef, viewId]);
}
```

export function useHandleTrigger(elementRef: RefObject<HTMLElement | null>, viewId: string | null): void {

## 9. 风险 + 开放问题

### 9.1 slash menu 触发字符 / 误触

V1 经验:slash 触发**严格限行首/段首**(避免输入英文 URL `http://` 误触)。L5-B3.1 沿用。

### 9.2 IME(中文输入)兼容

V1 SlashMenu.tsx 监听 compositionupdate/compositionend 把预编辑文字反馈给菜单过滤。L5-B3.1 driver PM Plugin 的 query 来自 doc 实际内容(IME 期间不写入 doc),所以 query 滞后于用户输入 — 体验差但不致命。**L5-B3.1 沿用 PM 标准行为,不做 IME 实时 query 反馈**(L5-C+ 真有用户反馈再加)。

### 9.3 handle widget 实施细节

handle 是 PM widget decoration 还是外挂 React 组件?

**A. widget decoration**(driver 内部完整 — 对齐 V2 driver 自治原则)
**B. NodeView 模式**(每个 block 用 NodeView 包,内部加 handle DOM)— V1 没用,因为 V1 textBlock 没 NodeView
**C. 外挂 React 组件**(V1 模式 — handle 在 EditorView 之外,通过 PM coords 计算位置)

**我推荐 A** — driver 自治 + PM 标准做法。实施细节实施时再深化。

### 9.4 dnd source/target 协议

dnd capability 协议跟 HTML5 native drag-and-drop 怎么协调?

**应对**:driver 内部用 native drag(HTML5 dragstart/dragover/drop),event handler 内调 `dnd.emit()` 同步走 capability 协议。**两套并行,native 是底层实现,capability 是协议层**。

### 9.5 floating-toolbar 跟 slash menu 同时显示

用户选中文字时输入 `/`,同时显两个浮层?

**应对**:slash menu 触发时光标 collapse(empty selection)— floating-toolbar 自动隐(它要求非空选区)。所以**互斥天然成立**。

### 9.6 handle 拖拽时 floating-toolbar / slash 等浮层

拖拽时浮层应该全关:dnd.emit('dnd.started') 时统一关所有交互浮层。

**实施**:WorkspaceInstance.tsx 加全局监听 dnd.started → 关 contextMenu/slash/handle/floatingToolbar 浮层。

### 9.7 context menu Turn Into 5 项太挤

V1 用子菜单(Turn Into ▶ → 浮 4 项)。L5-B3.1 严守 Q3=A 扁平 4 项。

**应对**:如视觉太挤,L5-B3.x 加子菜单支持(ContextMenuItem 加 children 字段)。

### 9.8 selection capability 性能

L5-B2 加了 Snapshot diff 真变才 emit。L5-B3.1 floating-toolbar 订阅同款 channel,不需重复优化。

### 9.9 dnd 拖拽视觉反馈

拖拽时 dropTarget 应该高亮(行间显示蓝色横线,Notion-style)。

**应对**:dnd.emit('dnd.over', { target, valid }) 时 driver 通过 widget decoration 显示行间横线。L5-B3.1 实施。

### 9.10 maxWidth 与 NavSide 折叠的兼容

NavSide 折叠时 SlotArea 全宽,NoteView maxWidth 仍居中(留白增多)— 符合预期。
